$serialNumber = "{diskSerialNumber}"
$disk = Get-Disk -SerialNumber $serialNumber

# Clear the disk
Clear-Disk -Number $disk.Number `
    -RemoveData `
    -Confirm:$false `
    -Verbose

# Verify the disk is clean
if ($disk.NumberOfPartitions -eq 0) {
    Write-Host "Disk is clean."
} else {
    Write-Host "Disk is not clean."
    exit 1
}
